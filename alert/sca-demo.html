Semua Shorthand
// Dialog (isToast = false, default) SCA.success('Berhasil!', 'Pesan...'); SCA.error('Gagal!', 'Error...'); SCA.warning('Peringatan!', '...'); SCA.info('Info', '...'); // Toast (isToast = true) SCA.success('OK!', 'Tersimpan.', true); SCA.error('Gagal!', '...', true, 'bottom-right'); // Selalu dialog + Promise SCA.confirm('Yakin?', 'Lanjutkan proses ini?').then(ok => { ... }); SCA.deleteConfirm('Nama Item').then(ok => { ... }); // Loading & close SCA.loading({ title: 'Memproses...' }); SCA.close(); // Special dialogs SCA.activateUser({ name: 'Budi', email: 'budi@pms.id', role: 'Manager' }).then(...); SCA.sendEmail({ to: 'user@pms.id', subject: 'Subjek' }).then(...); SCA.resetPassword({ userName: 'Budi', userEmail: 'budi@pms.id' }).then(...);
